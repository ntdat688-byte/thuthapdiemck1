``` ### Lưu ý quan trọng cho Backend (Google Apps Script) Vì bạn đã thêm cột "Khối" vào dữ liệu gửi đi, bạn cần **cập nhật lại code trong Google Apps Script** để nó lưu đúng cột này vào Sheet. Hãy sửa lại hàm `doPost` trong Apps Script như sau: ```javascript function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rawData = JSON.parse(e.postData.contents); var timestamp = new Date().toLocaleString("vi-VN"); // Trường hợp 1: Nhập lẻ (Single) if (rawData.type === 'single' || !rawData.type) { sheet.appendRow([ rawData.timestamp || timestamp, rawData.studentName, rawData.school, rawData.grade, // THÊM CỘT KHỐI Ở ĐÂY rawData.subject, rawData.score, rawData.teacherName, rawData.campus ]); } // Trường hợp 2: Nhập hàng loạt từ Excel (Bulk) else if (rawData.type === 'bulk' && rawData.data && rawData.data.length > 0) { var rowsToAdd = rawData.data.map(function(item) { return [ item.timestamp || timestamp, item.studentName, item.school, item.grade, // THÊM CỘT KHỐI Ở ĐÂY item.subject, item.score, item.teacherName, item.campus ]; }); var lastRow = sheet.getLastRow(); sheet.getRange(lastRow + 1, 1, rowsToAdd.length, rowsToAdd[0].length).setValues(rowsToAdd); } // Trường hợp 3: Gửi Link Sheet (Link) else if (rawData.type === 'link') { sheet.appendRow([ rawData.timestamp || timestamp, "LINK GOOGLE SHEET", rawData.url, "-", // Khối "-", // Môn "-", // Điểm rawData.teacherName, "-" ]); } return ContentService.createTextOutput(JSON.stringify({"result":"success"})) .setMimeType(ContentService.MimeType.JSON); }
